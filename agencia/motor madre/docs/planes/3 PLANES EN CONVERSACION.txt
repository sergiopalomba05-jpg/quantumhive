from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import torch

# Supongamos que importamos la librería oficial/comunitaria de F5-TTS
# Para instalar: pip install f5-tts fastapi uvicorn pydantic
from f5_tts.api import F5TTS

app = FastAPI(title="Motor de Voz Ultra-Realista (Estilo ElevenLabs)")

# Inicializar el modelo globalmente en la GPU Nvidia
device = "cuda" if torch.cuda.is_available() else "cpu"
print(f"Cargando modelo de voz en: {device}")
tts_model = F5TTS(model_type="F5-TTS", device=device)

class VoiceRequest(BaseModel):
    text: str
    voice_id: str       # ID del avatar o del cliente
    use_cache: bool = False # True para pre-renderizar videos del frontend

# Función generadora para hacer Streaming de audio (Evita latencia)
def generate_audio_stream(text: str, ref_audio_path: str):
    # El modelo procesa el texto y arroja los bytes de audio en trozos (chunks)
    # F5-TTS permite iterar sobre la marcha mediante generadores de Python
    chunks = tts_model.infer_stream(
        gen_text=text,
        ref_audio=ref_audio_path,
        ref_text="", # Opcional: el texto que se dice en el audio de referencia
    )
    for chunk in chunks:
        yield chunk # Envía los bytes crudos (PCM/WAV) al cliente de inmediato

@app.post("/v1/voice/generate")
async def generate_voice(request: VoiceRequest):
    # 1. Buscar el audio de referencia de 5 segundos (el clon)
    ref_audio_path = f"voices/{request.voice_id}_ref.wav"
    
    if not os.path.exists(ref_audio_path):
        raise HTTPException(status_code=404, detail="La voz especificada no existe.")

    # Caso de Uso A: Es para el caché del frontend (Videos pre-grabados)
    if request.use_cache:
        cache_path = f"cache/{request.voice_id}_{hash(request.text)}.wav"
        if os.path.exists(cache_path):
            # Si ya se generó antes, lo devolvemos directo del disco
            return StreamingResponse(open(cache_path, "rb"), media_type="audio/wav")
        
        # Si no existe, lo genera completo una sola vez y lo guarda
        tts_model.infer_to_file(gen_text=request.text, ref_audio=ref_audio_path, output_path=cache_path)
        return StreamingResponse(open(cache_path, "rb"), media_type="audio/wav")

    # Caso de Uso B: Servicio SaaS tipo ElevenLabs en tiempo real (Streaming puro)
    return StreamingResponse(
        generate_audio_stream(request.text, ref_audio_path),
        media_type="audio/wav"
    )

class ChatRequest(BaseModel):

@app.post("/v1/chat/completions")
async def chat_unificado(request: ChatRequest, authorization: str = Header(None)):
    validar_cliente_y_creditos(authorization)

    # 1. RUTA HACIA CHATGPT (OPENAI)
    if request.model.startswith("gpt"):
        client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
        
        def stream_openai():
            response = client.chat.completions.create(
                model=request.model, messages=request.messages, stream=True, temperature=request.temperature
            )
            for chunk in response:
                yield f"data: {chunk.model_dump_json()}\n\n"
        return StreamingResponse(stream_openai(), media_type="text/event-stream")

    # 2. RUTA HACIA CLAUDE (ANTHROPIC)
    elif request.model.startswith("claude"):
        client = anthropic.Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))
        
        def stream_anthropic():
            # Traducimos el formato de OpenAI al formato que entiende Anthropic
            with client.messages.stream(
                model=request.model, max_tokens=1024, messages=request.messages, temperature=request.temperature
            ) as stream:
                for text in stream.text_stream:
                    # Envolvemos la respuesta para que imite el formato estándar de OpenAI
                    yield f"data: {{\"choices\": [{{ \"delta\": {{ \"content\": \"{text}\" }} }}]}}\n\n"
        return StreamingResponse(stream_anthropic(), media_type="text/event-stream")

    # 3. RUTA HACIA GROK (X.AI)
    elif request.model.startswith("grok"):
        # El SDK de OpenAI es compatible con Grok cambiando la URL base
        client = OpenAI(api_key=os.getenv("XAI_API_KEY"), base_url="https://x.ai")
        
        def stream_grok():
            response = client.chat.completions.create(
                model=request.model, messages=request.messages, stream=True, temperature=request.temperature
            )
            for chunk in response:
                yield f"data: {chunk.model_dump_json()}\n\n"
        return StreamingResponse(stream_grok(), media_type="text/event-stream")

    # 4. RUTA HACIA META LLAMA LOCAL (Tu propio servidor vLLM con GPU)
    elif request.model.startswith("llama"):
        # Apuntas a la IP interna de tu máquina virtual con GPU de Google Cloud
        client = OpenAI(api_key="tu-token-interno", base_url="http://IP_DE_TU_VM_GPU:8000/v1")
        
        def stream_local():
            response = client.chat.completions.create(
                model="meta-llama/Llama-3.1-8B-Instruct", messages=request.messages, stream=True, temperature=request.temperature
            )
            for chunk in response:
                yield f"data: {chunk.model_dump_json()}\n\n"
        return StreamingResponse(stream_local(), media_type="text/event-stream")

    else:
        raise HTTPException(status_code=400, detail="Modelo no soportado o no implementado.")
